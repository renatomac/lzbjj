✅ 2️⃣ PDF View (REAL PDF)
views.py
from django.http import HttpResponse
from django.template.loader import render_to_string
from weasyprint import HTML
from .models import WaiverSignature


def waiver_pdf(request, pk):
    signature = WaiverSignature.objects.select_related(
        "waiver_version"
    ).get(pk=pk)

    html_string = render_to_string(
        "waiver/pdf.html",
        {
            "signature": signature,
            "waiver": signature.waiver_version,
        }
    )

    pdf = HTML(
        string=html_string,
        base_url=request.build_absolute_uri("/")
    ).write_pdf()

    response = HttpResponse(pdf, content_type="application/pdf")
    response["Content-Disposition"] = (
        f'inline; filename="waiver-{signature.id}.pdf"'
    )

    return response


✔ base_url is critical for CSS, images, logos
✔ inline displays in browser (change to attachment if you want download)

✅ 3️⃣ PDF Template (Print-Optimized)
templates/waiver/pdf.html
<!DOCTYPE html>
<html>
<head>
  <meta charset="utf-8">

  <style>
    @page {
      size: Letter;
      margin: 1in;
    }

    body {
      font-family: Arial, sans-serif;
      font-size: 12px;
      line-height: 1.6;
      color: #000;
    }

    h2, h3, h4 {
      margin-top: 20px;
    }

    .footer {
      margin-top: 40px;
      font-size: 10px;
    }
  </style>
</head>
<body>

  {{ waiver.content|safe }}

  <div class="footer">
    <hr>
    <p><strong>Participant:</strong> {{ signature.participant_full_name }}</p>
    <p><strong>Signed At:</strong> {{ signature.signed_at }}</p>
    <p><strong>IP Address:</strong> {{ signature.ip_address }}</p>
    <p><strong>User Agent:</strong> {{ signature.user_agent }}</p>
    <p><strong>Waiver Version:</strong> {{ waiver.version }}</p>
  </div>

</body>
</html>
